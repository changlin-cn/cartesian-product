# cartesian-product
# Usage

// npm install cartesian-product-fn --save
var cartesianProduct = require("cartesian-product-fn").default;

console.log(cartesianProduct({ a: [1, 2, 3], b: [4, 5, 6] }));
/*
[ [ 1, 4 ],

  [ 1, 6 ],
  [ 2, 4 ],
  [ 2, 5 ],
  [ 2, 6 ],
  [ 3, 4 ],
  [ 3, 5 ],
  [ 3, 6 ] ]
*/
